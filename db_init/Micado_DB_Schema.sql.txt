-- Name: active_features; Type: VIEW; Schema: micadoapp; Owner: -

CREATE VIEW active_features AS

CREATE VIEW document_type_translated AS
 SELECT dt.id,
    dt.icon,
    dt.issuer,
    dt.model,
    dt.validable,
    dt.validity_duration,
    dtt.lang,
    dtt.document,
    dtt.description,
    dtt.template_image
   FROM document_type dt,
    document_type_translation dtt
  WHERE (dt.id = dtt.id);
